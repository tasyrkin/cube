getData = (url, cb) ->
    http.get url, (res) ->

        output = ''

        res.on 'data', (chunk) ->
            output += chunk

        res.on 'end', () ->
            obj = JSON.parse output
            log 'Fetched ', obj.length, ' items.'
            cb obj

log 'Fetching data from ', url

deleteAllData url, () ->

addAllData = () ->
    getData url, (data) ->
        log 'Adding entries...'

        _.each data, (item, i, o) ->
            item = newItem item

            solrClient.add [ item ], (err, result) ->
                throw err if err
                if i is o.length-1
                    log 'Added ', i+1, ' entries.'
                    solrClient.commit () ->
                        return process.exit(0)

newItem = (srcItem) ->
    item = {}
    _.each schema, (property) ->
        item[property.id] = srcItem[property.id]

        if srcItem["organization"]
            org = srcItem["organization"].split('/').slice(2)
            return unless org.length
            team = []
            team.push org[0]
            team.push org.join('/') if org.length > 1
            item["team"] = team
    item.id = generateId()
    return item
