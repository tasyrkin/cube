solrClient.autoCommit = dbSettings.autoCommit

main = () =>
    getData deployctl_url, (deployctl_data) ->
        getData cube_url, (cube_data) ->
            cube_data = cube_data.response.docs
            _.each deployctl_data, (project) ->
                setId project, cube_data
            deleteAllData () ->
                addAllData deployctl_data

setId = (project, cube_data) ->
    _.each cube_data, (cp) ->
        cpgroup = cp.group[cp.group.length-1]
        prgroup = project.group[project.group.length-1]
        if cp.name is project.name and cpgroup is prgroup
            project.id = cp.id

addAllData = (data) ->
    _.each data, (item, i, o) ->
        item = newItem item

        solrClient.add [ item ], (err, result) ->
            throw err if err
            if i is o.length-1
                log 'Added ', i+1, ' entries.'
                solrClient.commit () ->
                    return process.exit(0)

newItem = (srcItem) ->
    item = {}
    _.each schema, (property) ->
        return unless srcItem[property.id]
        item[property.id] = srcItem[property.id]
        sep = property.separator || '/'

        if property.type is "facet" and typeof srcItem[property.id] is "string"
            item[property.id] = []
            _.each srcItem[property.id].split(','), (field, i) =>
                _.each field.split(sep), (token, i) =>
                    path = field.split(sep).slice(0, i).join(sep)
                    index = item[property.id].indexOf(path)
                    item[property.id].push path if index is -1 and path
                item[property.id].push field if field
        if srcItem["organization"]
            org = srcItem["organization"].split(sep).slice(2)
            return unless org.length
            team = []
            team.push org[0]
            team.push org.join(sep) if org.length > 1
            item["team"] = team
    item = solrManager.addObjSuffix 'project', item
    item.id = if srcItem.id then srcItem.id else generateId()
    return item

deleteAllData = (cb) ->

    query = solrClient.createQuery()
        .q("*:*")
        .start(0)
        .rows(1000)

    solrClient.deleteByQuery "*:*", (err, result) ->
        log 'Deleted all entries'
        cb()

getData = (url, cb) ->
    http.get url, (res) ->

        output = ''

        res.on 'data', (chunk) ->
            output += chunk

        res.on 'end', () ->
            obj = JSON.parse output
            log 'Fetched ', obj.length, ' items.'
            cb obj

generateId = () ->
    chars = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz"
    today = new Date()
    result = today.valueOf().toString 16
    result += chars.substr Math.floor(Math.random() * chars.length), 1
    result += chars.substr Math.floor(Math.random() * chars.length), 1
    return result
