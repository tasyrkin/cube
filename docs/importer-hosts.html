getData = (options, cb) ->
    http.get options, (res) ->

        output = ''

        res.on 'data', (chunk) ->
            output += chunk

        res.on 'end', () ->
            obj = JSON.parse output
            log 'Fetched ', obj.length, ' items.'
            cb obj

log 'Fetching data from ', options.path

deleteAllData () ->
    addAllData()

addAllData = () ->
    options.path = url
    getData options, (data) ->
        options.path = lifecycleids
        getData options, (lifecycles) ->
            options.path = hostroleids
            getData options, (hostroles) ->
                options.path = datacenterids
                getData options, (datacenters) ->

                    log 'Adding entries...'

                    _.each data, (item, i, o) ->
                        item = newItem(item, lifecycles,
                            hostroles, datacenters, data)

                        solrClient.add [ item ], (err, result) ->
                            throw err if err
                            if i is o.length-1
                                log 'Added ', i+1, ' entries.'
                                solrClient.commit () ->
                                    return process.exit(0)

resolveId = (id, key, data) ->
    resolved = null
    _.each data, (d) ->
        resolved = d[key] if d.id is id
    resolved

newItem = (srcItem, lc, hr, dc, src) ->
    item = {}
    _.each schema, (property) ->
        return if property.id is 'memory_range'
        return unless srcItem[property.id] or
            srcItem.physical_machine?[property.id]
        item[property.id] = srcItem[property.id]
        if property.id is 'assigned_memory'
            item.memory_range = getMemoryRange srcItem[property.id]
            item[property.id] = (srcItem[property.id] / 1048576).toFixed(2)
        if property.id is 'lifecycle_status_id'
            val = resolveId srcItem[property.id], 'name', lc
            item[property.id] = if val then val else srcItem[property.id]
        if property.id is 'host_role_id'
            val = resolveId srcItem[property.id], 'name', hr
            item[property.id] = if val then val else srcItem[property.id]
        if property.id is 'parent_host_id'
            val = resolveId srcItem[property.id], 'hostname', src
            item[property.id] = if val then val else srcItem[property.id]
        if property.id is 'data_center'
            dcid = srcItem.physical_machine?['data_center_id']
            if !dcid
                pid = srcItem['parent_host_id']
                _.each src, (s) ->
                    dcid = s.physical_machine?.data_center_id if s.id is pid
            val = resolveId dcid, 'name', dc
            item[property.id] = val
            delete item[property.id] unless val
        if property.id is 'model'
            item[property.id] = srcItem.physical_machine?['model']
        if property.id is 'created' or property.id is 'last_audited' or
           property.id is 'last_modified'
            [d, t] = srcItem[property.id].split(' ')
            date = new Date("#{d} #{t}")
            item[property.id] = date.toISOString()
    item = solrManager.addObjSuffix 'hosts', item
    item.id = srcItem.id
    return item

getMemoryRange = (mem) ->
    return null unless mem
    if mem <= 15000000 then return "0 to 15G"
    if mem > 15000000 and mem <= 30000000 then return "> 15G to 30G"
    if mem > 30000000 and mem <= 60000000 then return "> 30G to 60G"
    if mem > 60000000 and mem <= 90000000 then return "> 60G to 90G"
    return "> 90G"
